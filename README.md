# random-js

```js
let random = ['Александр Е. получил робота', 'Алёна С. получила робота', 'Дмитрий В. получил робота', 'Денис Т. получил робота', 'Сергей М. получил робота', 'Кирилл С. получил робота', 'Иван Г. получил робота', 'Татьяна Л. получила робота'];

let randomIndex = Math.floor((Math.random() * random.length) + 0);

let randomValue = random[randomIndex];

console.log(randomValue);
```
